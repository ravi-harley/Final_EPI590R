Objective 4

Write and use a function that does something with the data (1 pt)

new_std <- function(x) {
  n <- length(x)
  mean_val <- sum(x, na.rm = TRUE) / n
  variance <- sum((x - mean_val)^2, na.rm = TRUE) / (n - 1)
  std2 <- sqrt(variance)
  return(std2)  
}

x<- olympics$medal_num
